SECTION 2: ENUMS
================================================================

ENUM: user_role
  VALUES: user | volunteer | police | admin | guardian

ENUM: gender
  VALUES: female | male | non_binary | prefer_not_to_say

ENUM: alert_status
  VALUES: pending | active | accepted | resolved | false_alarm | escalated | cancelled

ENUM: alert_type
  VALUES: harassment | assault | medical_emergency | kidnapping_risk |
          cyberstalking | suspicious_activity | general_danger | stalking | theft

ENUM: sos_trigger_method
  VALUES: tap | long_press | voice | shake | silent | hidden_trigger | auto_journey

ENUM: volunteer_status
  VALUES: available | busy | offline | suspended

ENUM: verification_status
  VALUES: pending | verified | rejected | suspended

ENUM: journey_status
  VALUES: active | completed | cancelled | deviation_detected | sos_triggered | delayed

ENUM: report_category
  VALUES: unsafe_area | stalking | broken_streetlight | suspicious_behavior |
          unsafe_transport | harassment | poor_lighting | other

ENUM: evidence_type
  VALUES: audio | video | image | gps_log | screenshot

ENUM: notification_type
  VALUES: sos_alert | journey_alert | volunteer_alert | guardian_alert |
          community_alert | system_alert | escalation_alert

ENUM: route_risk_level
  VALUES: safe | low | moderate | high | critical

ENUM: incident_severity
  VALUES: low | moderate | high | critical

ENUM: fake_call_status
  VALUES: scheduled | active | completed | cancelled

ENUM: cyber_report_type
  VALUES: cyberstalking | fake_profile | blackmail | scam | harassment | doxxing

ENUM: transport_alert_type
  VALUES: route_deviation | suspicious_stoppage | speed_anomaly | unknown_vehicle

----------------------------------------------------------------
TABLE: users
----------------------------------------------------------------
PURPOSE: All platform users regardless of role

COLUMNS:
  id                    UUID            PRIMARY KEY DEFAULT uuid_generate_v4()
  full_name             VARCHAR(100)    NOT NULL
  email                 VARCHAR(150)    UNIQUE NOT NULL
  phone                 VARCHAR(15)     UNIQUE NOT NULL
  password_hash         TEXT            NOT NULL                  → bcrypt hashed
  role                  user_role       DEFAULT 'user'
  gender                gender
  date_of_birth         DATE
  profile_image_url     TEXT
  is_active             BOOLEAN         DEFAULT true
  is_verified           BOOLEAN         DEFAULT false             → OTP verified
  is_phone_verified     BOOLEAN         DEFAULT false
  is_email_verified     BOOLEAN         DEFAULT false
  last_login_at         TIMESTAMP
  created_at            TIMESTAMP       DEFAULT NOW()
  updated_at            TIMESTAMP       DEFAULT NOW()
  deleted_at            TIMESTAMP                                 → soft delete

INDEXES:
  idx_users_email       ON email
  idx_users_phone       ON phone
  idx_users_role        ON role

TABLE: sos_alerts
----------------------------------------------------------------
PURPOSE: Core SOS emergency records — most critical table

COLUMNS:
  id                    UUID            PRIMARY KEY DEFAULT uuid_generate_v4()
  alert_code            VARCHAR(20)     UNIQUE NOT NULL           → e.g. RK2024XXXX
  user_id               UUID            NOT NULL REFERENCES users(id)
  trigger_method        sos_trigger_method        NOT NULL
  alert_type            alert_type      DEFAULT 'general_danger'
  status                alert_status    DEFAULT 'pending'
  severity              incident_severity         DEFAULT 'high'
  description           TEXT
  trigger_location      GEOGRAPHY(POINT, 4326)    NOT NULL        → exact SOS trigger point
  trigger_address       TEXT                                      → reverse geocoded address
  current_location      GEOGRAPHY(POINT, 4326)                    → updated in real-time
  assigned_volunteer_id UUID            REFERENCES volunteers(id)
  assigned_police_id    UUID            REFERENCES police_accounts(id)
  assigned_station_id   UUID            REFERENCES police_stations(id)
  ai_classification     alert_type                                → AI classified type
  ai_confidence_score   DECIMAL(5,4)                              → 0.0 to 1.0
  ai_risk_score         DECIMAL(5,4)                              → 0.0 to 1.0
  escalated_at          TIMESTAMP
  escalation_reason     TEXT
  resolved_at           TIMESTAMP
  resolution_notes      TEXT
  volunteer_eta_seconds INTEGER
  police_eta_seconds    INTEGER
  is_test_alert         BOOLEAN         DEFAULT false
  created_at            TIMESTAMP       DEFAULT NOW()
  updated_at            TIMESTAMP       DEFAULT NOW()

INDEXES:
  idx_sos_alerts_user_id        ON user_id
  idx_sos_alerts_status         ON status
  idx_sos_alerts_location       ON trigger_location USING GIST
  idx_sos_alerts_created_at     ON created_at DESC

VIEW: v_daily_sos_stats
PURPOSE: Daily SOS metrics for admin dashboard
QUERY LOGIC:
  SELECT
    DATE(created_at) as date,
    COUNT(*) as total_alerts,
    COUNT(*) FILTER (WHERE status = 'resolved') as resolved,
    COUNT(*) FILTER (WHERE status = 'escalated') as escalated,
    AVG(EXTRACT(EPOCH FROM (resolved_at - created_at))) as avg_resolution_secs,
    COUNT(DISTINCT user_id) as unique_users
  FROM sos_alerts
  GROUP BY DATE(created_at)

VIEW: v_volunteer_performance
PURPOSE: Volunteer response rate dashboard
QUERY LOGIC:
  SELECT
    v.id,
    u.full_name,
    v.total_responses,
    v.successful_responses,
    v.rating,
    ROUND(v.successful_responses::decimal / NULLIF(v.total_responses,0) * 100, 2) as success_rate_pct
  FROM volunteers v
  JOIN users u ON v.user_id = u.id

VIEW: v_hotspot_heatmap
PURPOSE: Safety heatmap for map rendering
QUERY LOGIC:
  SELECT
    id,
    location,
    city,
    category,
    risk_score,
    report_count,
    peak_danger_hours
  FROM safety_hotspots
  WHERE is_active = true
  ORDER BY risk_score DESC

VIEW: v_active_volunteers_nearby
PURPOSE: Find available volunteers near a point (used in API)
QUERY LOGIC:
  Accepts: (lat FLOAT, lng FLOAT, radius_km INT)
  SELECT
    v.id,
    u.full_name,
    u.phone,
    ul.location,
    ST_Distance(ul.location, ST_MakePoint(lng, lat)::geography) as distance_meters
  FROM volunteers v
  JOIN users u ON v.user_id = u.id
  JOIN user_locations ul ON u.id = ul.user_id
  WHERE v.status = 'available'
    AND v.verification_status = 'verified'
    AND ST_DWithin(ul.location, ST_MakePoint(lng, lat)::geography, radius_km * 1000)
  ORDER BY distance_meters ASC
